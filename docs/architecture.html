WP_Query Integration

Use standard WordPress query arguments. No learning curve for new SQL syntax.

functions.php
use NCR\Core\Relationship;

// 1. Register a relationship between 'Movies' and 'Actors'
add_action('init', function() {
    Relationship::register([
        'id'          => 'movie_actors',
        'from'        => 'movie',
        'to'          => 'actor',
        'cardinality' => 'many-to-many', // Supports 1-to-1, 1-to-many
        'meta'        => true // Allow metadata on the connection itself
    ]);
});

// 2. Query Movies featuring a specific Actor
$query = new WP_Query([
    'post_type' => 'movie',
    'ncr_relationship' => [
        'id' => 'movie_actors',
        'related_to' => 1054 // Actor ID
    ]
]);
